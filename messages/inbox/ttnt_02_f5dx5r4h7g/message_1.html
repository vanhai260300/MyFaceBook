n = np.size(x)
w1 = (np.sum(np.multiply(x, y)) - 1/n*np.sum(x)*np.sum(y)) / (np.sum(np.square(x)) - 1/n*np.square(sum(x)))
return w1;

n = np.size(x)
w1 = getW1(x, y)
w0 = 1/n * np.sum(y) - w1*1/n*np.sum(x)
